" tabs and spaces
set shiftwidth=4
set tabstop=4
set softtabstop=4
set sw=4
set expandtab
set autoindent
set smartindent
filetype plugin indent on

" display
set ruler
set number
set hlsearch
set showmatch
set cin
syntax on
colorscheme desert

" control
set mouse=a
set nocompatible
set whichwrap+=<,>,h,l

" file encoding
set encoding=utf-8
set fencs=utf-8,ucs-bom,shift-jis,gb18030,gbk,gb2312,cp936
set fileencoding=utf-8


augroup filetype
  au! BufRead,BufNewFile,BufEnter *Makefile*,*makefile*,*.mk set filetype=make
augroup END
" In Makefiles, don't expand tabs to spaces, since we need the actual tabs
autocmd FileType make set noexpandtab
